994static GPtrArray *
995sanitize (MsdXrandrManager *manager, GPtrArray *array)
996{
997 int i;
998 GPtrArray *new;
999
1000 g_debug ("before sanitizing");
1001
1002 for (i = 0; i < array->len; ++i) {
1003 if (array->pdata[i]) {
1004 print_configuration (array->pdata[i], "before");
1005 }
1006 }
1007
1008
1009 /* Remove configurations that are duplicates of
1010 * configurations earlier in the cycle
1011 */
1012 for (i = 0; i < array->len; i++) {
1013 int j;
1014
1015 for (j = i + 1; j < array->len; j++) {
1016 MateRRConfig *this = array->pdata[j];
1017 MateRRConfig *other = array->pdata[i];
1018
1019 if (this && other && mate_rr_config_equal (this, other)) {
1020 g_debug ("removing duplicate configuration");
1021 g_object_unref (this);
1022 array->pdata[j] = NULL((void*)0);
1023 break;
1024 }
1025 }
1026 }
1027
1028 for (i = 0; i < array->len; ++i) {
1029 MateRRConfig *config = array->pdata[i];
1030
1031 if (config && config_is_all_off (config)) {
1032 g_debug ("removing configuration as all outputs are off");
1033 g_object_unref (array->pdata[i]);
1034 array->pdata[i] = NULL((void*)0);
1035 }
1036 }
1037
1038 /* Do a final sanitization pass. This will remove configurations that
1039 * don't fit in the framebuffer's Virtual size.
1040 */
1041
1042 for (i = 0; i < array->len; i++) {
1043 MateRRConfig *config = array->pdata[i];
1044
1045 if (config) {
1046 GError *error;
1047
1048 error = NULL((void*)0);
1049 if (!mate_rr_config_applicable (config, manager->priv->rw_screen, &error)) { /* NULL-GError */
1050 g_debug ("removing configuration which is not applicable because %s", error->message);
1051 g_error_free (error);
1052
1053 g_object_unref (config);
1054 array->pdata[i] = NULL((void*)0);
1055 }
1056 }
1057 }
1058
1059 /* Remove NULL configurations */
1060 new = g_ptr_array_new ();
1061
1062 for (i = 0; i < array->len; ++i) {
1063 if (array->pdata[i]) {
1064 g_ptr_array_add (new, array->pdata[i]);
1065 print_configuration (array->pdata[i], "Final");
1066 }
1067 }
1068
1069 if (new->len > 0) {
1070 g_ptr_array_add (new, NULL((void*)0));
1071 } else {
1072 g_ptr_array_free (new, TRUE(!(0)));
1073 new = NULL((void*)0);
1074 }
1075
1076 g_ptr_array_free (array, TRUE(!(0)));
1077
1078 return new;
1079}
